---
title: Get Documents
description: Retrieve a list of documents in your Devscribe workspace
api: GET /documents
num: 2
---

### Authorizations

<ParameterField name="Authorization" type="string" required>
  Bearer authentication header of the form `Bearer <token>`.
</ParameterField>

### Query Parameters

<ParameterField name="limit" type="integer (int32)">
  The maximum number of results to return
</ParameterField>

### Responses

<APIResponse status="200" description="Document response" contentType="application/json" defaultOpen>
  <APIResponseField name="name" type="string" description="The name of the document" />
  <APIResponseField name="tag" type="string" description="Tag to specify the type" />
</APIResponse>

<APIResponse status="400" description="Unexpected error" contentType="application/json">
  <APIResponseField name="error" type="integer" description="" />
  <APIResponseField name="message" type="string" description="" />
</APIResponse>
